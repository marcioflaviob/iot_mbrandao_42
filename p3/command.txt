#!/bin/bash

# Create k3d cluster with port mapping for the application
k3d cluster create mycluster \
  --port 8888:8888@loadbalancer \
  --port 8080:80@loadbalancer

# Wait for cluster to be ready
echo "Waiting for cluster to be ready..."
kubectl wait --for=condition=Ready nodes --all --timeout=60s

echo "Cluster created successfully!"
kubectl get nodes

#!/bin/bash

# Create argocd namespace
kubectl create namespace argocd

# Install ArgoCD
kubectl apply -n argocd -f https://raw.githubusercontent.com/argoproj/argo-cd/stable/manifests/install.yaml

# Wait for ArgoCD to be ready
echo "Waiting for ArgoCD to be ready..."
kubectl wait --for=condition=available --timeout=300s deployment/argocd-server -n argocd

# Get initial admin password
echo "ArgoCD is ready!"
echo "Initial admin password:"
kubectl -n argocd get secret argocd-initial-admin-secret -o jsonpath="{.data.password}" | base64 -d
echo ""

# Port-forward ArgoCD server (run in background)
echo "Setting up port-forward for ArgoCD UI on http://localhost:8080"
kubectl port-forward svc/argocd-server -n argocd 8080:443 &

echo "ArgoCD UI will be available at https://localhost:8080"
echo "Username: admin"
